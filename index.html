function doGet(e) { const codeToFind = e.parameter.code; const sheet = SpreadsheetApp.getActiveSpreadsheet().getSheets()[0]; const data = sheet.getDataRange().getValues(); const headers = data[0]; for (let i = 1; i < data.length; i++) { // Ahora busca en las columnas A (0), B (1) o C (2) // que corresponden a EAN, DUN y Cod HBC if (data[i][0].toString() == codeToFind || data[i][1].toString() == codeToFind || data[i][2].toString() == codeToFind) { let result = {}; headers.forEach((header, index) => { if (header) { result[header] = data[i][index] || "---"; } }); return ContentService.createTextOutput(JSON.stringify({ status: "ok", data: result })).setMimeType(ContentService.MimeType.JSON); } } return ContentService.createTextOutput(JSON.stringify({ status: "error", msg: "Código no encontrado" })).setMimeType(ContentService.MimeType.JSON); }
